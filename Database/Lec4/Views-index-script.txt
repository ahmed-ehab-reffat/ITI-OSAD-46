-- -------------- DB Views ----------------------------
-- db view : select statement stored in the db as an object
-- how to create view
create or replace view emps_depts_view
as
select employee_id, last_name, salary, email, hire_date, 
	employees.department_id, department_name
from employees, departments
where employees.department_id = departments.department_id;

-- calling view
select employee_id, last_name, email, department_id
from emps_depts_view
where department_id = 90;

-- + ve
-- 1. reusability 
-- 2. make complex queries easy 
-- 3. performance [ No +ve or -ve performance on using the view - it is about the same as the original SELECT. ]
-- 4. security ****

-- Views Types  المتعارف عليه 
-- 1. Simple View [ from 1 table, no functions, no group by : Allow DMLs ]
-- 2. Complex View [ from more than 1 table, may contain functions, may contain group by,
--   				cannot have dmls ]

create or replace view simple_vu20
as select * from employees
where department_id = 20
with check option;
-- with read only;

update simple_vu20
set salary = salary + 1000;

update simple_vu20
set department_id = 30;

-- ### views types
-- 	[ Standard View -  Materialized View ( store data ) used in data warehouse ]
